CREATE DATABASE wccmmud;

CREATE TABLE gossip (
	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	name CHAR(32) NOT NULL,
	text TINYTEXT NOT NULL,
	timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE = MyISAM;

CREATE TABLE online (
	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user CHAR(32) NOT NULL,
	busy VARCHAR(1) DEFAULT '-' NOT NULL,
	timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE = MyISAM;

CREATE TABLE gangs (
	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	gang CHAR(32) NOT NULL,
	leader CHAR(32),
	members TINYINT(4) DEFAULT 1,
	exp BIGINT(20) DEFAULT 0,
	expold BIGINT(20) DEFAULT 0,
	creation VARCHAR(9),
	timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE = MyISAM;

CREATE TABLE users (
	id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	user CHAR(32) NOT NULL,
	class CHAR(10),
	title CHAR(16),
	exp BIGINT(13) DEFAULT 0,
	expold BIGINT(13) DEFAULT 0,
	gang CHAR(32),
	timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE = MyISAM;
